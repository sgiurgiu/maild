cmake_minimum_required(VERSION 3.2)
project(maild)

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/")
set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++1y -Wall -pedantic -Wextra -fPIC")

SET(MAILD_VERSION_MAJOR "0")
SET(MAILD_VERSION_MINOR "0")
SET(MAILD_VERSION_PATCH "1")

add_definitions(-DMAILD_VERSION="${MAILD_VERSION_MAJOR}.${MAILD_VERSION_MINOR}.${MAILD_VERSION_PATCH}")

if (NOT CMAKE_BUILD_TYPE)
    message(STATUS "No build type selected, default to Release")
    set(CMAKE_BUILD_TYPE "Release")
endif()

include(GNUInstallDirs)
include(CMakePackageConfigHelpers)


add_subdirectory(lib)
add_subdirectory(app)
add_subdirectory(test)
add_subdirectory(web)

SET(CPACK_PACKAGE_NAME "maild")
SET(CPACK_PACKAGE_DESCRIPTION_SUMMARY "MailD")
SET(CPACK_RESOURCE_FILE_LICENSE_PROVIDED FALSE)
SET(CPACK_INTERACTIVE FALSE)
SET(CPACK_INCLUDE_SUBDIR TRUE)
SET(CPACK_PACKAGE_VERSION_MAJOR ${MAILD_VERSION_MAJOR})
SET(CPACK_PACKAGE_VERSION_MINOR ${MAILD_VERSION_MINOR})
SET(CPACK_PACKAGE_VERSION_PATCH ${MAILD_VERSION_PATCH})


set(CPACK_RPM_RELOCATION_PATHS "/${CMAKE_INSTALL_SYSCONFDIR}" "${CMAKE_INSTALL_BINDIR}")
set(CPACK_INCLUDE_TOPLEVEL_DIRECTORY 0) # don't prepend package name
#set(CPACK_PACKAGING_INSTALL_PREFIX ${CMAKE_INSTALL_PREFIX})
set(CPACK_RPM_POST_INSTALL_SCRIPT_FILE "${CMAKE_CURRENT_SOURCE_DIR}/postinstall")

install(DIRECTORY "www/"
  DESTINATION "/var/www/html/maild")

install(FILES "conf/maild.conf.sample"
  DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/maild")

install(FILES "conf/log.properties.sample"
  DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/maild")
  
install(FILES "conf/maild_web.conf.sample"
  DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/maild"
  )

install(FILES "conf/log.properties.sample"
  DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/maild"
  )
install(FILES "conf/web_log.properties.sample"
  DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/maild"
  )

install(FILES "service/maild.service" 
    DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/systemd/system"  
    )

install(FILES "service/maild_web.service" 
    DESTINATION "/${CMAKE_INSTALL_SYSCONFDIR}/systemd/system"    
    )
    
SET(CPACK_PACKAGE_FILE_NAME "${CPACK_PACKAGE_NAME}-${CPACK_PACKAGE_VERSION_MAJOR}.${CPACK_PACKAGE_VERSION_MINOR}.${CPACK_PACKAGE_VERSION_PATCH}")
list(APPEND CPACK_RPM_EXCLUDE_FROM_AUTO_FILELIST_ADDITION "/var")
list(APPEND CPACK_RPM_EXCLUDE_FROM_AUTO_FILELIST_ADDITION "/etc/systemd")
list(APPEND CPACK_RPM_EXCLUDE_FROM_AUTO_FILELIST_ADDITION "/etc/systemd/system")


INCLUDE(CPack)
